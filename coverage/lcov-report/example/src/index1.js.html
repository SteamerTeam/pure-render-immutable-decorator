import React, { Component } from 'react';
import { render } from 'react-dom';
import pureRender from '../../src/';
import Immutable from 'immutable';
 
 
@pureRender
export class List extends Component {
	constructor(props, context) {
		super(props, context);
 
		this.state = {
			subtitle: "Strength",
			heroes: Immutable.fromJS([
				{
					id: 1,
					name: "Earthshaker"
				},
				{
					id: 2,
					name: "Sven"
				},
				{
					id: 3,
					name: "Tiny"
				},
				{
					id: 4,
					name: "Kunkka"
				},
			])
		};
 
	}
 
	shouldComponentUpdate() {
		return true;
	}
 
	componentDidMount() {
		
	}
 
	render() {
 
		let {
			title,
		} = this.props;
 
		let {
			subtitle,
		} = this.props;
 
		let heroes = this.props.heroes || [];
 
		return (
			<div>
				<p>{title}</p>
				<p>{subtitle}</p>
		        <ul>
		        	{
		        		heroes.map((item) => {
		        			return (
		        				<li key={item.get('id')}>{item.get('name')}</li>
		        			);
		        		})
		        	}
		        </ul>
		        <p>{this.state.subtitle}</p>
		        <ul>
		        	{
		        		this.state.heroes.map((item) => {
		        			return (
		        				<li key={item.get('id')}>{item.get('name')}</li>
		        			);
		        		})
		        	}
		        </ul>
		    </div>
		);
	}
}
